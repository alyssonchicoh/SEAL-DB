SELECT client_name, date, amount, product_name, price,  employee_name

FROM   sales, client, employee,product 

WHERE 
             employee_id = employee_id_fk and 
             product_id = product_id_fk  and
             client_id =  client_id_fk

ORDER BY  product_name;
